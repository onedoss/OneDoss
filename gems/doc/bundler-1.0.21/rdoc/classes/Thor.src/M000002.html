    def register(klass, subcommand_name, usage, description, options={})
      if klass <= Thor::Group
        desc usage, description, options
        define_method(subcommand_name) { invoke klass }
      else
        desc usage, description, options
        subcommand subcommand_name, klass
      end
    end
